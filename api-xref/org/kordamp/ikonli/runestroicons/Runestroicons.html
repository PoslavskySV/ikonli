553     public static Runestroicons findByDescription(String description) {
554         for (Runestroicons font : values()) {
555             if (font.getDescription().equals(description)) {
556                 return font;
557             }
558         }
559         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
560     }
