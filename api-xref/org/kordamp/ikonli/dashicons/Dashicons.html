283     public static Dashicons findByDescription(String description) {
284         for (Dashicons font : values()) {
285             if (font.getDescription().equals(description)) {
286                 return font;
287             }
288         }
289         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
290     }
291 
292     private String description;
293     private char code;
294 
295     Dashicons(String description, char code) {
296         this.description = description;
297         this.code = code;
298     }
299 
300     @Override
301     public String getDescription() {
302         return description;
303     }
304 
305     @Override
306     public char getCode() {
307         return code;
308     }
