199     public static Octicons findByDescription(String description) {
200         for (Octicons font : values()) {
201             if (font.getDescription().equals(description)) {
202                 return font;
203             }
204         }
205         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
206     }
207 
208     private String description;
209     private char code;
210 
211     Octicons(String description, char code) {
212         this.description = description;
213         this.code = code;
214     }
215 
216     @Override
217     public String getDescription() {
218         return description;
219     }
220 
221     @Override
222     public char getCode() {
223         return code;
224     }
