158     public static Feather findByDescription(String description) {
159         for (Feather font : values()) {
160             if (font.getDescription().equals(description)) {
161                 return font;
162             }
163         }
164         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
165     }
